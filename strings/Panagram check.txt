#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main()
{
    cin.sync_with_stdio(false); 
    cin.tie(NULL);    
    string str;
    getline(cin,str);
    int n=str.length();
    int mark[26]={0};
    int index=0;
    int flag=0;
    for(int i=0;i<n;i++)
    {
        if(str[i]>='A' && str[i]<='Z')
            index=str[i]-'A';
            
        else if(str[i]>='a' && str[i]<='z')    
            index=str[i]-'a';
            
        else
          continue;
          
            
         mark[index]='1';
        
    }
    
    for(int i=0;i<26;i++)
    {
        if(mark[i]==0)
        {
            cout<<"NO"<<"\n";
            flag=1;
            break;
        }
        
    }
    
    if(flag==0)
    cout<<"YES"<<"\n";
}
